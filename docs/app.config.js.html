        .primaryPalette('indigo')

    $mdThemingProvider.theme('theme-view')
        .primaryPalette('blue')
        .accentPalette('red')

    $mdThemingProvider.theme('theme-about')
        .primaryPalette('green')

    $mdThemingProvider.theme('theme-login')

})
